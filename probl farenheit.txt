 public class Prbl1 {

private static final double MINC = -273.15;
private static final double MINF = -459.67;

public static double transformaInGradeCelsius(double F) {
return 5.0 / 9.0 * (F - 32.0);
}

public static double transformaInGradeFahrenheit(double C) {
return 9.0 / 5.0 * C + 32.0;
}

public static void main(String[] args) {
Scanner scan = new Scanner(System.in);
double C, F;
do {
System.out.print("Introduceti prima valoare in grade celsius (mai mare decat -273.15): ");
C = scan.nextDouble();
} while (C < MINC);
do {
System.out.print("Introduceti prima valoare in grade fahrenheit (mai mare decat -459.67): ");
F = scan.nextDouble();
} while (F < MINF);

System.out.println(C + " grade Celsius = " + transformaInGradeFahrenheit(C) + " grade Fahrenheit");
System.out.println(F + " grade Fahrenheit = " + transformaInGradeCelsius(F) + " grade Celsius");
}

}
